 41	def scale_theme(self, theme:dict):
 42		"do necessary rescaling math for drawing"
 43		if "title_font" in theme:
 44			theme["title_font"] = pygame.font.SysFont(None, int(145 * self.scale))
 45		if "title_font_size" in theme:
 46			theme["title_font_size"] = int(theme["title_font_size"] * self.scale)
 47		if "border_width" in theme:

275	def mainloop(self, window:pygame.Surface):
276		"draw the window and redirect events"
277		self.loop = True
278		while self.loop:
279			self.clock.tick(30)
280			if self.main_menu.is_enabled():
281				self.main_menu.draw(window)
282				self.main_menu.update(pygame.event.get())
283				if self.main_menu._current is self.replay_menu:
284					window.blit(self.info_surf, (0 * self.scale, 250 * self.scale))
285				pygame.display.update()
286		self.games = None
287		if self.player1 == 1:
288			self.player1_name = "CP" + str(self.cp1_diff)
289		if self.player2 == 1:
290			self.player2_name = "CP" + str(self.cp2_diff)
